闭包
    全局变量和局部变量
    正常情况下函数内部可以读取全局变量   函数外部不能读取局部变量
    1、概念
        能够读取其他函数内部变量的函数
    2、用途
        读取函数内部变量
        变量始终保持在内存中
    3、缺点
        造成内存泄漏
        变量篡改

(function main(){
    var name = 1
    var _name = 1
    function a() {
        console.log(name)
        name = 2
        _name = 2
    }

    function b(){
        console.log(name)
    }
})()
